3. Easing

Move the mouse across the screen and the symbol will follow.

import
  p5

var x = 1.0
var y = 1.0
var easing = 0.05
proc setup() {.exportc.} =
  createCanvas(400, 400)
  noStroke()

proc draw() {.exportc.} =
  background(237, 34, 93)
  let targetX = mouseX
  let dx = targetX - x
  x += dx * easing
  let targetY = mouseY
  let dy = targetY - y
  y += dy * easing
  ellipse(x, y, 66, 66)
