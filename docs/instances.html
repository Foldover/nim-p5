8. Instances

This program creates a canvas in a local p5 instance using "instance mode", which allows to place a P5 instance into a specific <div> if desired. Otherwise this is the same example as get_started.


import
  jsffi, p5

proc foo(): int =
  400

instance:
  let x = 5
  let y = 10
  setup:
    let bar = foo()
    createCanvas(bar, 400)
    background(200)
  draw:
    if mouseIsPressed:
      fill(0)
    else:
      let fillBy = 255
      fillBy.fill
    mouseX.ellipse(mouseY, 40, 40)
